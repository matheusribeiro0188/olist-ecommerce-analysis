-- Faturamento por estado

SELECT 
    c.customer_state,
    COUNT(DISTINCT o.order_id) AS total_pedidos,
    ROUND(SUM(i.price), 2) AS faturamento_total,
    ROUND(AVG(i.price), 2) AS ticket_medio
FROM olist_customers_dataset c
INNER JOIN olist_orders_dataset o ON o.customer_id = c.customer_id
INNER JOIN olist_order_items_dataset i ON i.order_id = o.order_id
WHERE o.order_status = 'delivered'
GROUP BY c.customer_state
ORDER BY faturamento_total DESC;

-- Faturamento por categoria

SELECT 
    t.string_field_1 AS categoria,
    COUNT(DISTINCT i.order_id) AS total_pedidos,
    ROUND(SUM(i.price), 2) AS faturamento_total,
    ROUND(AVG(i.price), 2) AS ticket_medio
FROM olist_order_items_dataset i
INNER JOIN olist_products_dataset p ON p.product_id = i.product_id
INNER JOIN product_category_name_translation t ON t.string_field_0 = p.product_category_name
GROUP BY t.string_field_1
ORDER BY faturamento_total DESC
LIMIT 10;

-- Top 10 vendedores por faturamento

SELECT
    i.seller_id,
    s.seller_city,
    s.seller_state,
    COUNT(DISTINCT i.order_id) AS total_pedidos,
    ROUND(SUM(i.price), 2) AS faturamento_total,
    ROUND(AVG(i.price), 2) AS ticket_medio
FROM olist_order_items_dataset i
INNER JOIN olist_sellers_dataset s ON s.seller_id = i.seller_id
GROUP BY i.seller_id, s.seller_city, s.seller_state
ORDER BY faturamento_total DESC
LIMIT 10;

-- Tempo médio de entrega por estado (em dias)

SELECT
    c.customer_state,
    COUNT(DISTINCT o.order_id) AS total_pedidos,
    ROUND(AVG(
        JULIANDAY(o.order_delivered_) - 
        JULIANDAY(o.order_purchase_t)
    ), 1) AS media_dias_entrega
FROM olist_orders_dataset o
INNER JOIN olist_customers_dataset c ON c.customer_id = o.customer_id
WHERE o.order_status = 'delivered'
AND o.order_delivered_ IS NOT NULL
GROUP BY c.customer_state
ORDER BY media_dias_entrega ASC;
